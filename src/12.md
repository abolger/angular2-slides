## Using `@Input`

```typescript
import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-say-hello',
  templateUrl: './hello.component.html'
})
export class SayHelloComponent {
  @Input()
  public name: string;
}
```

```html
<h3 *ngIf="name">Hello {{name}}!</h3>
```

```html
<app-say-hello name="John"></app-say-hello>
```
